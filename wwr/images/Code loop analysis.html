Final Version (with hints)

python
# Word guessing game - Final Version

secret_word = "mosiah"
guess = ""
guess_count = 0

print("Welcome to the word guessing game!\n")

# display initial hint
print("Your hint is:", "_ " * len(secret_word))

while guess != secret_word:
    guess = input("What is your guess? ").lower()
    guess_count += 1

    # Check length
    if len(guess) != len(secret_word):
        print("Sorry, the guess must have the same number of letters as the secret word.")
        continue  # go back to the top of the loop

    # If correct
    if guess == secret_word:
        print("Congratulations! You guessed it!")
        break

    # Generate hint
    hint = []
    for i in range(len(secret_word)):
        if guess[i] == secret_word[i]:
            hint.append(guess[i].upper())      # correct spot
        elif guess[i] in secret_word:
            hint.append(guess[i].lower())      # present but wrong spot
        else:
            hint.append("_")                   # not present
    print("Your hint is:", " ".join(hint))

print(f"It took you {guess_count} guesses.")
